DROP DATABASE IF EXISTS JDBCTEST;
GRANT USAGE ON *.* TO 'JDBCTEST'@'%' IDENTIFIED BY 'JDBCTEST';
GRANT USAGE ON *.* TO 'JDBCTEST'@'localhost' IDENTIFIED BY 'JDBCTEST';
DROP USER 'JDBCTEST'@'%';
DROP USER 'JDBCTEST'@'localhost';
CREATE DATABASE IF NOT EXISTS JDBCTEST DEFAULT CHARACTER SET 'utf8' DEFAULT COLLATE 'utf8_bin';
CREATE USER 'JDBCTEST'@'%' IDENTIFIED BY 'JDBCTEST';
CREATE USER 'JDBCTEST'@'localhost' IDENTIFIED BY 'JDBCTEST';
GRANT ALL ON JDBCTEST.* TO 'JDBCTEST'@'%' WITH GRANT OPTION;
GRANT ALL ON JDBCTEST.* TO 'JDBCTEST'@'localhost' WITH GRANT OPTION;
