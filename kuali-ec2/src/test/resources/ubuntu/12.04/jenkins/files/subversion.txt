function install_subversion {

SVN_USERNAME=user
SVN_PASSWORD="password"
REPO="https://svn.kuali.org/repos/rice/sandbox/devops/ci/svn-credentials-caching-dir"

#Check for subversion install
if hash svn 2>/dev/null; then
  echo "Subversion is already installed."
else
  echo "Subversion is not currently installed.  Installing..."
  apt-get $SILENT install subversion
  svn co $REPO
  # Allow for storing password plain text
  eval "sed -i -e '/store-plaintext-passwords \= no/astore-plaintext-passwords \= yes' /root/.subversion/servers"
  # Create a temporary directory in a repo to set authentication.
  svn mkdir --username $SVN_USERNAME --password $SVN_PASSWORD --parents "$REPO/mytmp" --message "Testing authentication"
  # Remove temporary directory in repo
  svn delete "$REPO/mytmp" --message "Removing mytmp dir"
fi
}
