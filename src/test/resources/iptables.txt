# Redirect all internally generated traffic from port 80 to port 8080
iptables -t nat -A OUTPUT -p tcp -o lo --dport 80 -j DNAT --to 127.0.0.1:8080

# Redirect all externally generated traffic from port 80 to port 8080
iptables --table nat --append PREROUTING --protocol tcp --dport 80 --jump REDIRECT --to-port 8080

# Delete an OUTPUT rule
iptables --table nat --delete OUTPUT 1

# List all iptables rules
iptables --table nat --numeric --list --line-numbers
